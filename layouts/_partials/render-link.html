{{- /* Render a link with automatic detection of external links. Parameters (dict): - href: The URL - text: The link text - title: (optional) Title attribute - private: (optional) Mark as private link External links get target="_blank" and rel="external noopener noreferrer". Private links get an additional "(besloten omgeving)" screen-reader label. Source: adapted from MinBZK/moza-site (layouts/_partials/render-link.html) */ -}} {{- $u := urls.Parse .href -}} {{- $isExternal := and $u.IsAbs (hasPrefix $u.Scheme "http") -}} {{- if .private -}} {{ .text }} (besloten omgeving) {{- else if $isExternal -}} {{ .text }} {{- else -}} {{ .text }} {{- end -}}
